import pandas as pd
from scipy.stats import f_oneway

# Assuming your DataFrame is named df

# Group the response times by complaint type
grouped = df.groupby('Complaint Type')['Request_Closing_Time'].apply(list)

# Convert timedelta to hours for better interpretation
grouped_hours = [pd.Series(times).dt.total_seconds() / 3600 for times in grouped]

# Perform one-way ANOVA
f_stat, p_value = f_oneway(*grouped_hours)

# Display results
print(f"F-statistic: {f_stat}")
print(f"P-value: {p_value}")

# Interpretation
if p_value < 0.05:
    print("Reject the Null Hypothesis: Average response time differs across complaint types.")
else:
    print("Fail to Reject the Null Hypothesis: No significant difference in average response time across complaint types.")
